嗯对。我们考虑字符串的前缀数组：$nxt_i$ 表示 $s_{0\sim i}$ 的最大真前缀字符串长度使得这个真前缀字符串等于等长的后缀字符串。不存在为 $0$。易得 $nxt_0=0$ 。

好的 $O(n^3)$ 听取 TLE 声一片，

然后我们注意到一个事实：$nxt_i(i\ge 1)\le nxt_{i-1}+1$ 。\
为什么？使用反证法证明。首先有 $nxt_i\le i$ 。如果 $nxt_i\ge nxt_{i-1}+2$ ，那么 $s_{0\sim nxt_i}=s_{i-nxt_i+1\sim i}$ ，所以 $s_{0\sim nxt_i-1}=s_{i-nxt_i+1\sim i-1}$ ，故 $nxt_{i-1}\ge nxt_i-1\ge nxt_{i-1}+2-1=nxt_{i-1}+1$ 不成立。$\Box$ 。


```cpp
string s;
for(int i=1;i<s.size();i++){
	for(int j=nxt[i-1]-1;j>0;j--)
		if(s.substr(0,j)==s.substr(i-j+1,j)){
			nxt[i]=j;
			break;
		}
}
```

所以最多积累 $n-2$ 次，消耗 $n-1$ 次，总字符串比较次数为 $O(n)$ ，复杂度 $O(n^2)$ 。

```cpp
string s;
for(int i=1;i<s.size();i++){
	int j=nxt[i-1];

	nxt[i]=j+(s[j]==s[i]);
```

KMP 适用于什么题呢？字符串匹配！比如说“在 $t$ 中找到 $s$ 出现了多少次，分别在哪里”之类的。\
怎么做呢？将 $s$ 和 $t$ 接在一起，中间隔个字符，这个字符**需要 $\boldsymbol s$ 和 $\boldsymbol t$ 中都没有出现过**，然后对新串 KMP，对于 $i\ge|s|$，如果 $nxt_i=s$ 的话，那么在 $t$ 中 $i-2|s|+1$ 处出现了 $s$ 。

## I

```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
string s,t;
string g;
const int maxn=1e6+5,mod=1e9+7;
int nxt[maxn],tmp[maxn];
ll ans=1;
ll cnt(ll x){
	if(!x)return 0;
	if(tmp[x])return tmp[x];
	return tmp[x]=cnt(nxt[x-1])+1;
}
int main(){
	ios::sync_with_stdio(false);
	cin.tie(nullptr),cout.tie(nullptr);
	int t;
	cin>>t;
	while(t--){
		cin>>g;
		fill(tmp+1,tmp+1+g.size(),0);
		for(int i=1;i<g.size();i++){
			int j=nxt[i-1];
			while(j!=0&&g[i]!=g[j])j=nxt[j-1];
			nxt[i]=j+(g[i]==g[j]);
		}
		for(int i=0;i<g.size();i++)cnt(nxt[i]);
		for(int i=g.size()-1;i>=0;i--)
			while(nxt[i]*2>i+1)nxt[i]=nxt[nxt[i]-1];
		for(int i=0;i<g.size();i++)ans*=(cnt(nxt[i])+1),ans%=mod;
		cout<<ans<<"\n";ans=1;
	}
	return 0;
}
```
第一版 TLE 代码 ↑ $O(n^2)$ 。

```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
string s,t;
string g;
const int maxn=1e6+5,mod=1e9+7;
int nxt[maxn],tmp[maxn],dep[maxn],fa[25][maxn];
ll ans=1;
ll cnt(ll x){
	if(!x)return 0;
	if(tmp[x])return tmp[x];
	return tmp[x]=cnt(nxt[x-1])+1;
}
int main(){
	ios::sync_with_stdio(false);
	cin.tie(nullptr),cout.tie(nullptr);
	int t;
	cin>>t;
	while(t--){
		cin>>g;
		fill(tmp+1,tmp+1+g.size(),0);
		for(int i=1;i<g.size();i++){
			int j=nxt[i-1];
			while(j!=0&&g[i]!=g[j])j=nxt[j-1];
			nxt[i]=j+(g[i]==g[j]);
		}
		for(int i=0;i<g.size();i++)cnt(nxt[i]);
		for(int i=0;i<g.size();i++){
			if(!nxt[i])continue;
			dep[nxt[i]]=dep[nxt[nxt[i]-1]]+1;
			fa[0][nxt[i]]=nxt[nxt[i]-1];
			for(int j=1;j<=20;j++)fa[j][nxt[i]]=fa[j-1][fa[j-1][nxt[i]]];
		}
		for(int i=0;i<g.size();i++){
			if(nxt[i]*2<=i+1)continue;
			for(int j=20;j>=0;j--)
				if(fa[j][nxt[i]]*2>i+1)nxt[i]=fa[j][nxt[i]];
			nxt[i]=fa[0][nxt[i]];
		}
		for(int i=0;i<g.size();i++)ans*=(cnt(nxt[i])+1),ans%=mod;
		cout<<ans<<"\n";ans=1;
	}
	return 0;
}
```

$O(n\log n)$ 。
